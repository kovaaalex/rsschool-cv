function descendingOrder(n){
    if (n < 0) return
    let result = ""
    let arrayOfDigits = Array.from(String(n), Number);
    arrayOfDigits = arrayOfDigits.sort((a, b) => a > b ? -1 : 1)
    for (let i of arrayOfDigits)
        result += i
    return +result
}
